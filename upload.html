// Acció d'enviar al Webhook de n8n uploadBtn.addEventListener('click', async () => { if (!selectedFile) return; uploadBtn.disabled = true; uploadBtn.classList.add('opacity-50', 'cursor-not-allowed'); progressContainer.classList.remove('hidden'); progressBar.classList.replace('bg-red-500', 'bg-purple-500'); progressBar.classList.replace('bg-green-500', 'bg-purple-500'); progressBar.style.width = '10%'; statusMsg.textContent = 'Convertint document...'; statusMsg.className = 'text-center text-sm text-purple-400 mt-2 font-semibold'; let p = 10; const interval = setInterval(() => { p = Math.min(p + Math.random() * 15, 85); progressBar.style.width = p + '%'; }, 300); try { // Convertir fitxer a base64 const base64 = await new Promise((resolve, reject) => { const reader = new FileReader(); reader.onload = () => resolve(reader.result); reader.onerror = reject; reader.readAsDataURL(selectedFile); }); statusMsg.textContent = 'Enviant a n8n...'; // Enviar en format que espera el webhook const res = await fetch(N8N_WEBHOOK, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ data: base64, mimeType: selectedFile.type, fileName: selectedFile.name }) }); clearInterval(interval); progressBar.style.width = '100%'; if (res.ok) { progressBar.classList.replace('bg-purple-500', 'bg-green-500'); statusMsg.textContent = '✅ Document guardat a Pinecone correctament!'; statusMsg.className = 'text-center text-sm text-green-400 mt-2 font-bold'; setTimeout(() => removeFile(), 4000); } else { throw new Error('Error al servidor n8n'); } } catch (err) { clearInterval(interval); progressBar.style.width = '100%'; progressBar.classList.replace('bg-purple-500', 'bg-red-500'); statusMsg.textContent = '❌ Error: ' + err.message; statusMsg.className = 'text-center text-sm text-red-400 mt-2 font-bold'; } finally { uploadBtn.disabled = false; uploadBtn.classList.remove('opacity-50', 'cursor-not-allowed'); } });
